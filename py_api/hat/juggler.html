 29async def connect(address: str, *,
 30                  autoflush_delay: typing.Optional[float] = 0.2,
 31                  ) -> 'Connection':
 32    """Connect to remote server
 33
 34    Address represents remote WebSocket URL formated as
 35    ``<schema>://<host>:<port>/<path>`` where ``<schema>`` is ``ws`` or
 36    ``wss``.
 37
 38    Argument `autoflush_delay` defines maximum time delay for automatic
 39    synchronization of `local_data` changes. If `autoflush_delay` is set to
 40    ``None``, automatic synchronization is disabled and user is responsible
 41    for calling :meth:`Connection.flush_local_data`. If `autoflush_delay` is
 42    set to ``0``, synchronization of `local_data` is performed on each change
 43    of `local_data`.
 44
 45    Args:
 46        address: remote server address
 47        autoflush_delay: autoflush delay
 48
 49    """
 50    session = aiohttp.ClientSession()
 51
 52    try:
 53        ws = await session.ws_connect(address, max_msg_size=0)
 54
 55    except BaseException:
 56        await aio.uncancellable(session.close())
 57        raise
 58
 59    return _create_connection(aio.Group(), ws, autoflush_delay, session)

 62async def listen(host: str,
 63                 port: int,
 64                 connection_cb: ConnectionCb, *,
 65                 ws_path: str = '/ws',
 66                 static_dir: typing.Optional[pathlib.Path] = None,
 67                 index_path: typing.Optional[str] = '/index.html',
 68                 pem_file: typing.Optional[pathlib.Path] = None,
 69                 autoflush_delay: typing.Optional[float] = 0.2,
 70                 shutdown_timeout: float = 0.1
 71                 ) -> 'Server':
 72    """Create listening server
 73
 74    Each time server receives new incomming juggler connection, `connection_cb`
 75    is called with newly created connection.
 76
 77    If `static_dir` is set, server serves static files is addition to providing
 78    juggler communication.
 79
 80    If `index_path` is set, request for url path ``/`` are redirected to
 81    `index_path`.
 82
 83    If `pem_file` is set, server provides `https/wss` communication instead
 84    of `http/ws` communication.
 85
 86    Argument `autoflush_delay` is associated with all connections associated
 87    with this server (see `connect`).
 88
 89    `shutdown_timeout` defines maximum time duration server will wait for
 90    regular connection close procedures during server shutdown. All connections
 91    that are not closed during this period are forcefully closed.
 92
 93    Args:
 94        host: listening hostname
 95        port: listening TCP port
 96        connection_cb: connection callback
 97        ws_path: WebSocket url path segment
 98        static_dir: static files directory path
 99        index_path: index path
100        pem_file: PEM file path
101        autoflush_delay: autoflush delay
102        shutdown_timeout: shutdown timeout
103
104    """
105    async_group = aio.Group()
106
107    server = Server()
108    server._connection_cb = connection_cb
109    server._autoflush_delay = autoflush_delay
110    server._async_group = async_group
111
112    routes = []
113
114    if index_path:
115
116        async def root_handler(request):
117            raise aiohttp.web.HTTPFound(index_path)
118
119        routes.append(aiohttp.web.get('/', root_handler))
120
121    routes.append(aiohttp.web.get(ws_path, server._ws_handler))
122
123    if static_dir:
124        routes.append(aiohttp.web.static('/', static_dir))
125
126    app = aiohttp.web.Application()
127    app.add_routes(routes)
128    runner = aiohttp.web.AppRunner(app)
129    await runner.setup()
130    async_group.spawn(aio.call_on_cancel, runner.cleanup)
131
132    try:
133        ssl_ctx = _create_ssl_context(pem_file) if pem_file else None
134        site = aiohttp.web.TCPSite(runner=runner,
135                                   host=host,
136                                   port=port,
137                                   shutdown_timeout=shutdown_timeout,
138                                   ssl_context=ssl_ctx,
139                                   reuse_address=True)
140        await site.start()
141
142    except BaseException:
143        await aio.uncancellable(async_group.async_close())
144        raise
145
146    return server

173def _create_connection(async_group, ws, autoflush_delay, session=None):
174    conn = Connection()
175    conn._async_group = async_group
176    conn._ws = ws
177    conn._autoflush_delay = autoflush_delay
178    conn._session = session
179    conn._remote_change_cbs = util.CallbackRegistry()
180    conn._remote_data = None
181    conn._local_data = None
182    conn._message_queue = aio.Queue()
183    conn._flush_queue = aio.Queue()
184    conn._local_data_queue = aio.Queue()
185
186    async_group.spawn(aio.call_on_cancel, conn._on_close)
187    async_group.spawn(conn._receive_loop)
188    async_group.spawn(conn._sync_loop)

193class Connection(aio.Resource):
194    """Connection
195
196    For creating new connection see `connect` coroutine.
197
198    """
199
200    @property
201    def async_group(self) -> aio.Group:
202        """Async group"""
203        return self._async_group
204
205    @property
206    def local_data(self) -> json.Data:
207        """Local data"""
208        return self._local_data
209
210    @property
211    def remote_data(self) -> json.Data:
212        """Remote data"""
213        return self._remote_data
214
215    def register_change_cb(self,
216                           cb: typing.Callable[[], None]
217                           ) -> util.RegisterCallbackHandle:
218        """Register remote data change callback"""
219        return self._remote_change_cbs.register(cb)
220
221    def set_local_data(self, data: json.Data):
222        """Set local data
223
224        Raises:
225            ConnectionError
226
227        """
228        try:
229            self._local_data_queue.put_nowait(data)
230            self._local_data = data
231
232        except aio.QueueClosedError:
233            raise ConnectionError()
234
235    async def flush_local_data(self):
236        """Force synchronization of local data
237
238        Raises:
239            ConnectionError
240
241        """
242        try:
243            flush_future = asyncio.Future()
244            self._flush_queue.put_nowait(flush_future)
245            await flush_future
246
247        except aio.QueueClosedError:
248            raise ConnectionError()
249
250    async def send(self, msg: json.Data):
251        """Send message
252
253        Raises:
254            ConnectionError
255
256        """
257        if self._async_group.is_closing:
258            raise ConnectionError()
259
260        await self._ws.send_json({'type': 'MESSAGE', 'payload': msg})
261
262    async def receive(self) -> json.Data:
263        """Receive message
264
265        Raises:
266            ConnectionError
267
268        """
269        try:
270            return await self._message_queue.get()
271
272        except aio.QueueClosedError:
273            raise ConnectionError()
274
275    async def _on_close(self):
276        self._message_queue.close()
277        self._flush_queue.close()
278        self._local_data_queue.close()
279
280        while not self._flush_queue.empty():
281            f = self._flush_queue.get_nowait()
282            if not f.done():
283                f.set_exception(ConnectionError())
284
285        await self._ws.close()
286        if not self._session:
287            return
288        await self._session.close()
289
290    async def _receive_loop(self):
291        try:
292            while True:
293                msg_ws = await self._ws.receive()
294                if self._ws.closed or msg_ws.type == aiohttp.WSMsgType.CLOSING:
295                    break
296                if msg_ws.type != aiohttp.WSMsgType.TEXT:
297                    raise Exception('unsupported message type')
298
299                msg = json.decode(msg_ws.data)
300
301                if msg['type'] == 'MESSAGE':
302                    self._message_queue.put_nowait(msg['payload'])
303
304                elif msg['type'] == 'DATA':
305                    self._remote_data = json.patch(self._remote_data,
306                                                   msg['payload'])
307                    self._remote_change_cbs.notify()
308
309                else:
310                    raise Exception("invalid message type")
311
312        except Exception as e:
313            mlog.error("juggler receive loop error: %s", e, exc_info=e)
314
315        finally:
316            self._async_group.close()
317
318    async def _sync_loop(self):
319        data = None
320        synced_data = None
321        flush_future = None
322
323        try:
324            if not self.is_open:
325                return
326            get_data_future = self._async_group.spawn(
327                self._local_data_queue.get)
328            get_flush_future = self._async_group.spawn(
329                self._flush_queue.get)
330
331            while self.is_open:
332
333                await asyncio.wait([get_data_future, get_flush_future],
334                                   return_when=asyncio.FIRST_COMPLETED)
335
336                if get_flush_future.done():
337                    flush_future = get_flush_future.result()
338                    get_flush_future = self._async_group.spawn(
339                        self._flush_queue.get)
340
341                else:
342                    await asyncio.wait([get_flush_future],
343                                       timeout=self._autoflush_delay)
344
345                    if get_flush_future.done():
346                        flush_future = get_flush_future.result()
347                        get_flush_future = self._async_group.spawn(
348                            self._flush_queue.get)
349                    else:
350                        flush_future = None
351
352                if get_data_future.done():
353                    data = get_data_future.result()
354                    get_data_future = self._async_group.spawn(
355                        self._local_data_queue.get)
356
357                if self._autoflush_delay != 0:
358                    if not self._local_data_queue.empty():
359                        data = self._local_data_queue.get_nowait_until_empty()
360
361                if synced_data is not data:
362                    diff = json.diff(synced_data, data)
363                    synced_data = data
364                    if diff:
365                        await self._ws.send_json({'type': 'DATA',
366                                                  'payload': diff})
367
368                if flush_future and not flush_future.done():
369                    flush_future.set_result(True)
370
371        except aio.QueueClosedError:
372            pass
373
374        except Exception as e:
375            mlog.error("juggler sync loop error: %s", e, exc_info=e)
376
377        finally:
378            self._async_group.close()

389class RpcConnection(aio.Resource):
390    """Remote procedure call juggler connection wrapper
391
392    Actions are functions/coroutines identified by action name and called with
393    arguments received as part of `rpc` message.
394
395    If `actions` dictionary doesn't contain requested action, `default_action`
396    is used - first argument passed to `default_action` is requested action
397    name and rest of arguments are ones received as part of `rpc` message.
398
399    If `actions` dictionary doesn't contain requested action and
400    `default_action` is not defined, rpc call results with exception
401    response.
402
403    """
404
405    def __init__(self,
406                 conn: Connection,
407                 actions: typing.Dict[str, aio.AsyncCallable] = {},
408                 default_action: typing.Optional[aio.AsyncCallable] = None):
409        self._conn = conn
410        self._actions = actions
411        self._default_action = default_action
412        self._message_queue = aio.Queue()
413        self._last_msg_id = 0
414        self._call_futures = {}
415
416        self.async_group.spawn(self._receive_loop)
417
418    @property
419    def async_group(self) -> aio.Group:
420        """Async group"""
421        return self._conn.async_group
422
423    @property
424    def local_data(self) -> json.Data:
425        """Local data"""
426        return self._conn.local_data
427
428    @property
429    def remote_data(self) -> json.Data:
430        """Remote data"""
431        return self._conn.remote_data
432
433    def register_change_cb(self,
434                           cb: typing.Callable[[], None]
435                           ) -> util.RegisterCallbackHandle:
436        """Register remote data change callback"""
437        return self._conn.register_change_cb(cb)
438
439    def set_local_data(self, data: json.Data):
440        """Set local data
441
442        Raises:
443            ConnectionError
444
445        """
446        self._conn.set_local_data(data)
447
448    async def flush_local_data(self):
449        """Force synchronization of local data
450
451        Raises:
452            ConnectionError
453
454        """
455        await self._conn.flush_local_data()
456
457    async def send(self, msg: json.Data):
458        """Send message
459
460        Raises:
461            ConnectionError
462
463        """
464        await self._conn.send(msg)
465
466    async def receive(self) -> json.Data:
467        """Receive message
468
469        Raises:
470            ConnectionError
471
472        """
473        try:
474            return await self._message_queue.get()
475
476        except aio.QueueClosedError:
477            raise ConnectionError()
478
479    async def call(self,
480                   action: str,
481                   *args: json.Data
482                   ) -> json.Data:
483        """Call remote action
484
485        Raises:
486            ConnectionError
487
488        """
489        if not self.is_open:
490            raise ConnectionError()
491
492        self._last_msg_id += 1
493        msg_id = self._last_msg_id
494        future = asyncio.Future()
495
496        try:
497            self._call_futures[msg_id] = future
498            await self.send({'type': 'rpc',
499                             'id': msg_id,
500                             'direction': 'request',
501                             'action': action,
502                             'args': list(args)})
503            return await future
504
505        finally:
506            del self._call_futures[msg_id]
507
508    async def _receive_loop(self):
509        try:
510            while True:
511                msg = await self._conn.receive()
512
513                if isinstance(msg, dict) and msg.get('type') == 'rpc':
514                    await self._process_msg_rpc(msg)
515
516                else:
517                    self._message_queue.put_nowait(msg)
518
519        except Exception as e:
520            mlog.error("rpc receive loop error: %s", e, exc_info=e)
521
522        finally:
523            self.close()
524            self._message_queue.close()
525            for future in self._call_futures.values():
526                if not future.done():
527                    future.set_exception(ConnectionError())
528
529    async def _process_msg_rpc(self, msg):
530        if msg['direction'] == 'request':
531            await self._process_msg_rpc_req(msg)
532
533        elif msg['direction'] == 'response':
534            await self._process_msg_rpc_res(msg)
535
536        else:
537            raise Exception('invalid rpc direction')
538
539    async def _process_msg_rpc_req(self, msg):
540        action = self._actions.get(msg['action'])
541        args = msg['args']
542
543        if not action and self._default_action:
544            action = functools.partial(self._default_action, msg['action'])
545
546        try:
547            if not action:
548                raise Exception('invalid action')
549            result = await aio.call(action, *args)
550            success = True
551
552        except Exception as e:
553            result = str(e)
554            success = False
555
556        await self.send({'type': 'rpc',
557                         'id': msg['id'],
558                         'direction': 'response',
559                         'success': success,
560                         'result': result})
561
562    async def _process_msg_rpc_res(self, msg):
563        future = self._call_futures.get(msg['id'])
564        if not future or future.done():
565            return
566
567        if msg['success']:
568            future.set_result(msg['result'])
569        else:
570            future.set_exception(Exception(msg['result']))
